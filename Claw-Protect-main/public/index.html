50+ Endpoints. One API Key.

Integrate any module into your pipeline with a single REST call.

scan-request.sh 
# Run a network scan via the Claw Protect API
curl -X POST https://api.clawprotect.dev/api/v1/scan \
  -H "Authorization: Bearer cp_live_k3y...r8nd0m" \
  -H "Content-Type: application/json" \
  -d '{
    "module": "network_scanner",
    "target": "192.168.1.0/24",
    "options": {
      "ports": "1-1024",
      "intensity": "thorough",
      "os_detection": true
    }
  }'

# Response
{
  "scan_id": "sc_8f3k2m9x",
  "status": "running",
  "module": "network_scanner",
  "hosts_discovered": 14,
  "eta_seconds": 42
}
